线性表是一种一对一的关系，树是一种一对多的关系，而图则是一种多对多的关系。实际上，我们将线性表和树看成是图的特殊情况，它们都可以是一种特殊的图。

1. 图的概念

通常用 V（Vertex）表示顶点的集合，用 E（Edge）表示边的集合，则图就是由一组 非空的 有限顶点集合 V 和一组有限边集合 E 组成，记为 G（V，E）。其中：

  • 边又分为 有向边 和 无向边，它们都可以表示为顶点对的形式，例如：
    • 无向边：（v，w）∈ E，其中 v，w ∈ V
    • 有向边：< v，w > ∈ E，其中 v，w ∈ V

另外，我们不考虑 重边 和 自回路 （这样的图称为简单图，我们只考虑这种图）。

  • 重边：我们默认只有一条边可以直接连通两个顶点（如果两个顶点之间有多条直接连通的线就不考虑）。
  • 自回路：一条边一定是从一个顶点指向另一个顶点，不可能指向自己。

概念解释
弧（Arc）边的另一种称呼
无向图（Digraph）图中所有的边没有特定的指向
有向图（Undigraph）图中所有的边是有方向的（有可能单向，有可能双向）
完全图任意两个顶点间都有边$\color{red}{直接}$相连
权重（Weight）与图的边有关的数，可能表示两个顶点之间的距离或者花费
顶点的度（Degree）对无向图，顶点所连接的边的数量
顶点的入度（Indegree）对有向图，指向顶点的边的数量
顶点的出度（Outdegree）对有向图，从顶点出发的边的数量
路径（Path）从一个顶点到另一个顶点的序列
路径长度路径上边的数目（如果带权，则是所有边的权重和）
起点等于终点的路径
连通图从一个顶点开始，可以达到图中任意一个其他顶点
非连通图图中存在不可达的顶点
连通分量对非连通图，它的极大连通子图称为连通分量
网络带权重的连通图
强连通有向图中两个顶点之间存在双向路径，则称这两个顶点强连通
强连通图有向图中任意两个顶点之间均能强连通
强连通分量有向图的极大强连通子图
弱连通如果将强连通图的方向抹除，它就变成连通图，这就是弱连通

关于图的操作集有很多，但最基本的如下：

图的抽象数据类型：

类型名称：图（Graph）

数据对象集：G（V，E）由一个非空的有限顶点集合 V 和一个有限边集合 E 组成。

基本操作集：

  • Graph Create()：建立并返回空图
  • Graph InsertVertex(Graph G, Vertex V)：将顶点 V 插入图 G
  • Graph InsertEdge(Graph G, Edge E)：将边 E 插入图 G
  • void DFS(Graph G, Vertex V)：从顶点 V 出发深度优先遍历图 G
  • void BFS(Graph G, Vertex V)：从顶点 V 出发广度优先遍历图 G
  • void ShortestPath(Graph G, Vertex V, int Dist[])：计算图 G 中顶点 V 到任意其它顶点的最短路径
  • void MST(Graph G)：计算图 G 的最小生成树

2. 图的表示

图有很多种表示方法，比如：邻接矩阵、邻接表、十字链表等，最常用的是邻接矩阵和邻接表。

2.1 邻接矩阵

通过邻接矩阵 $G[N] [N]$ 表示图，首先，将 N 个顶点从 0 到 N -1 进行编号，如果两个顶点之间有边相连，那么填入 1，如果两个顶点之间没有边相连则为 0。 $$ G[N][N] = \begin{cases} 1& 若是 G 中的边 \\ 0& 否则 \end{cases} $$ 下图为一个无向图的邻接矩阵表示。

在实际编程中，通常使用二维数组的形式存储。对于无向图而言，邻接矩阵是对称的，因此只需要通过存储左下三角矩阵或右上三角矩阵，即可确定整个图，进而节省一半的存储空间（矩阵压缩）。无向图某个顶点的度是其所对应一行（或列）非 0 元素的个数。

对于有向图，邻接矩阵不是对称的，所以我们不能采用这种方法来保存图。有向图的顶点所对应一行非 0 元素的个数被称为「出度」，对应一列非 0 元素的个数被称为「入度」。

对于网络，我们只需要 G [i] [j] 的值 1，换成它边的权重即可，如果没有边连接，为 $\infty$。

邻接矩阵的优缺点：

优点缺点
直观、简单、好理解浪费空间 —— 存稀疏图
（顶点多而边很少）
方便检查任意一对顶点之间是否存在边浪费时间 —— 统计稀疏图一共有多少边
方便查看任一顶点的「邻接点」（有边直接相连的顶点）
方便计算任一顶点的「度」

图的建立 - 邻接矩阵表示：

/* 图的邻接矩阵表示法 */
 
#define MaxVertexNum 100    /* 最大顶点数设为100 */
#define INFINITY 65535        /* ∞设为双字节无符号整数的最大值65535*/
typedef int Vertex;         /* 用顶点下标表示顶点,为整型 */
typedef int WeightType;        /* 边的权值设为整型 */
typedef char DataType;        /* 顶点存储的数据类型设为字符型 */
 
/* 边的定义 */
typedef struct ENode *PtrToENode;
struct ENode{
    Vertex V1, V2;      /* 有向边<V1, V2> */
    WeightType Weight;  /* 权重 */
};
typedef PtrToENode Edge;
        
/* 图结点的定义 */
typedef struct GNode *PtrToGNode;
struct GNode{
    int Nv;  /* 顶点数 */
    int Ne;  /* 边数   */
    WeightType G[MaxVertexNum][MaxVertexNum]; /* 邻接矩阵 */
    DataType Data[MaxVertexNum];      /* 存顶点的数据 */
    /* 注意：很多情况下，顶点无数据，此时Data[]可以不用出现 */
};
typedef PtrToGNode MGraph; /* 以邻接矩阵存储的图类型 */
 
 
 
MGraph CreateGraph( int VertexNum )
{ /* 初始化一个有VertexNum个顶点但没有边的图 */
    Vertex V, W;
    MGraph Graph;
     
    Graph = (MGraph)malloc(sizeof(struct GNode)); /* 建立图 */
    Graph->Nv = VertexNum;
    Graph->Ne = 0;
    /* 初始化邻接矩阵 */
    /* 注意：这里默认顶点编号从0开始，到(Graph->Nv - 1) */
    for (V=0; V < Graph->Nv; V++)
        for (W=0; W < Graph->Nv; W++)
            Graph->G[V][W] = INFINITY;
    return Graph; 
}
        
void InsertEdge( MGraph Graph, Edge E )
{
     /* 插入边 <V1, V2> */
     Graph->G[E->V1][E->V2] = E->Weight;    
     /* 若是无向图，还要插入边<V2, V1> */
     Graph->G[E->V2][E->V1] = E->Weight;
}
 
MGraph BuildGraph()
{
    MGraph Graph;
    Edge E;
    Vertex V;
    int Nv, i;
     
    scanf("%d", &Nv);   /* 读入顶点个数 */
    Graph = CreateGraph(Nv); /* 初始化有Nv个顶点但没有边的图 */ 
     
    scanf("%d", &(Graph->Ne));   /* 读入边数 */
    if ( Graph->Ne != 0 ) { /* 如果有边 */ 
        E = (Edge)malloc(sizeof(struct ENode)); /* 建立边结点 */ 
        /* 读入边，格式为"起点 终点 权重"，插入邻接矩阵 */
        for (i=0; i<Graph->Ne; i++) {
            scanf("%d %d %d", &E->V1, &E->V2, &E->Weight); 
            /* 注意：如果权重不是整型，Weight的读入格式要改 */
            InsertEdge( Graph, E );
        }
    } 
 
    /* 如果顶点有数据的话，读入数据 */
    for (V=0; V < Graph->Nv; V++) 
        scanf(" %c", &(Graph->Data[V]));
 
    return Graph;
}

2.2 邻接表

邻接表是将所有顶点用一个指针数组$G[N]$表示，每个元素表示一个节点，其值指向该顶点所有相邻顶点构成的链表（它的表示是不唯一的，因为在一串上面，这几个节点出现的顺序是无所谓的），一个有向图的邻接表示例如下

但是它真的很省空间吗？我们看 A 和 B 之间如果有一条边，那么在 A 这个链表上会有 B，B 这个链表上会有 A，这个图中的每一条边都被存了两次，而且在存一条边的时候，我们不仅要存一个整数（顶点的编号），我们还要有指针指向下一个，所以一条边占了差不多两个整数。对于网络来说，结构中还要增加权重的域。综上所述，邻接表只适用于稀疏图的情况。

图的建立 - 邻接表表示：

/* 图的邻接表表示法 */
 
#define MaxVertexNum 100    /* 最大顶点数设为100 */
typedef int Vertex;         /* 用顶点下标表示顶点,为整型 */
typedef int WeightType;        /* 边的权值设为整型 */
typedef char DataType;        /* 顶点存储的数据类型设为字符型 */
 
/* 边的定义 */
typedef struct ENode *PtrToENode;
struct ENode{
    Vertex V1, V2;      /* 有向边<V1, V2> */
    WeightType Weight;  /* 权重 */
};
typedef PtrToENode Edge;
 
/* 邻接点的定义 */
typedef struct AdjVNode *PtrToAdjVNode; 
struct AdjVNode{
    Vertex AdjV;        /* 邻接点下标 */
    WeightType Weight;  /* 边权重 */
    PtrToAdjVNode Next;    /* 指向下一个邻接点的指针 */
};
 
/* 顶点表头结点的定义 */
typedef struct Vnode{
    PtrToAdjVNode FirstEdge;/* 边表头指针 */
    DataType Data;            /* 存顶点的数据 */
    /* 注意：很多情况下，顶点无数据，此时Data可以不用出现 */
} AdjList[MaxVertexNum];    /* AdjList是邻接表类型 */
 
/* 图结点的定义 */
typedef struct GNode *PtrToGNode;
struct GNode{  
    int Nv;     /* 顶点数 */
    int Ne;     /* 边数   */
    AdjList G;  /* 邻接表 adjacent list */
};
typedef PtrToGNode LGraph; /* 以邻接表方式存储的图类型 */
 
 
 
LGraph CreateGraph( int VertexNum )
{ /* 初始化一个有VertexNum个顶点但没有边的图 */
    Vertex V;
    LGraph Graph;
     
    Graph = (LGraph)malloc( sizeof(struct GNode) ); /* 建立图 */
    Graph->Nv = VertexNum;
    Graph->Ne = 0;
    /* 初始化邻接表头指针 */
    /* 注意：这里默认顶点编号从0开始，到(Graph->Nv - 1) */
       for (V=0; V < Graph->Nv; V++)
        Graph->G[V].FirstEdge = NULL;
             
    return Graph; 
}
        
void InsertEdge( LGraph Graph, Edge E )
{
    PtrToAdjVNode NewNode;
     
    /* 插入边 <V1, V2> */
    /* 为V2建立新的邻接点 */
    NewNode = (PtrToAdjVNode)malloc(sizeof(struct AdjVNode));
    NewNode->AdjV = E->V2;
    NewNode->Weight = E->Weight;
    /* 将V2插入V1的表头 */
    NewNode->Next = Graph->G[E->V1].FirstEdge;
    Graph->G[E->V1].FirstEdge = NewNode;
         
    /* 若是无向图，还要插入边 <V2, V1> */
    /* 为V1建立新的邻接点 */
    NewNode = (PtrToAdjVNode)malloc(sizeof(struct AdjVNode));
    NewNode->AdjV = E->V1;
    NewNode->Weight = E->Weight;
    /* 将V1插入V2的表头 */
    NewNode->Next = Graph->G[E->V2].FirstEdge;
    Graph->G[E->V2].FirstEdge = NewNode;
}
 
LGraph BuildGraph()
{
    LGraph Graph;
    Edge E;
    Vertex V;
    int Nv, i;
     
    scanf("%d", &Nv);   /* 读入顶点个数 */
    Graph = CreateGraph(Nv); /* 初始化有Nv个顶点但没有边的图 */ 
     
    scanf("%d", &(Graph->Ne));   /* 读入边数 */
    if ( Graph->Ne != 0 ) { /* 如果有边 */ 
        E = (Edge)malloc( sizeof(struct ENode) ); /* 建立边结点 */ 
        /* 读入边，格式为"起点 终点 权重"，插入邻接矩阵 */
        for (i=0; i<Graph->Ne; i++) {
            scanf("%d %d %d", &E->V1, &E->V2, &E->Weight); 
            /* 注意：如果权重不是整型，Weight的读入格式要改 */
            InsertEdge( Graph, E );
        }
    } 
 
    /* 如果顶点有数据的话，读入数据 */
    for (V=0; V<Graph->Nv; V++) 
        scanf(" %c", &(Graph->G[V].Data));
 
    return Graph;
}

3. 图的遍历

图的遍历和树的遍历意思是一样的，图的遍历就是指将图中的每一个顶点都访问一遍，且不能有重复的访问。这里介绍两种遍历方法，分别是「DFS」（深度优先搜索 Depth First Search）、「BFS」（广度优先搜索 ）

3.1 深度优先搜索 DFS

/* 邻接表存储的图 - DFS */
 
void Visit( Vertex V )
{
    printf("正在访问顶点%d\n", V);
}
 
/* Visited[]为全局变量，已经初始化为false */
void DFS( LGraph Graph, Vertex V, void (*Visit)(Vertex) )
{   /* 以V为出发点对邻接表存储的图Graph进行DFS搜索 */
    PtrToAdjVNode W;
     
    Visit( V ); /* 访问第V个顶点 */
    Visited[V] = true; /* 标记V已访问 */
 
    for( W=Graph->G[V].FirstEdge; W; W=W->Next ) /* 对V的每个邻接点W->AdjV */
        if ( !Visited[W->AdjV] )    /* 若W->AdjV未被访问 */
            DFS( Graph, W->AdjV, Visit );    /* 则递归访问之 */
}

3.2 广度优先搜索 BFS

/* 邻接矩阵存储的图 - BFS */
 
/* IsEdge(Graph, V, W)检查<V, W>是否图Graph中的一条边，即W是否V的邻接点。  */
/* 此函数根据图的不同类型要做不同的实现，关键取决于对不存在的边的表示方法。*/
/* 例如对有权图, 如果不存在的边被初始化为INFINITY, 则函数实现如下:         */
bool IsEdge( MGraph Graph, Vertex V, Vertex W )
{
    return Graph->G[V][W]<INFINITY ? true : false;
}
 
/* Visited[]为全局变量，已经初始化为false */
void BFS ( MGraph Graph, Vertex S, void (*Visit)(Vertex) )
{   /* 以S为出发点对邻接矩阵存储的图Graph进行BFS搜索 */
    Queue Q;     
    Vertex V, W;
 
    Q = CreateQueue( MaxSize ); /* 创建空队列, MaxSize为外部定义的常数 */
    /* 访问顶点S：此处可根据具体访问需要改写 */
    Visit( S );
    Visited[S] = true; /* 标记S已访问 */
    AddQ(Q, S); /* S入队列 */
     
    while ( !IsEmpty(Q) ) {
        V = DeleteQ(Q);  /* 弹出V */
        for( W=0; W<Graph->Nv; W++ ) /* 对图中的每个顶点W */
            /* 若W是V的邻接点并且未访问过 */
            if ( !Visited[W] && IsEdge(Graph, V, W) ) {
                /* 访问顶点W */
                Visit( W );
                Visited[W] = true; /* 标记W已访问 */
                AddQ(Q, W); /* W入队列 */
            }
    } /* while结束*/
}
